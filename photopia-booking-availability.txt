// Virtual for next available date
availabilitySchema.virtual('nextAvailableDate').get(function() {
  const today = new Date()
  today.setHours(0, 0, 0, 0)
  
  // Check next 30 days
  for (let i = 0; i < 30; i++) {
    const checkDate = new Date(today)
    checkDate.setDate(today.getDate() + i)
    
    const dayOfWeek = checkDate.getDay()
    const dayName = ['sunday', 'monday', 'tuesday', 'wednesday', 'thursday', 'friday', 'saturday'][dayOfWeek]
    
    const daySchedule = this.defaultSchedule[dayName]
    
    if (daySchedule && daySchedule.isActive) {
      // Check if not in custom blocked dates
      const isBlocked = this.customDates.some(
        cd => cd.date.toDateString() === checkDate.toDateString() && 
              cd.type === 'blocked'
      )
      
      if (!isBlocked) {
        return checkDate
      }
    }
  }
  
  return null
})
